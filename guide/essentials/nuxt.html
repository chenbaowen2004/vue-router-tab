# Nuxt Config

nuxt.config.js

export default {
  // import routerTab plugin
  plugins: ['@/plugins/routerTab'],

  router: {
    extendRoutes(routes, resolve) {
      // add Iframe route
      routes.push({
        name: 'iframe',
        path: '/iframe/:src/:title?/:icon?',
        component: resolve(__dirname, 'pages/-Iframe.js'),
        props: true
      })
    }
  },

  build: {
    // Babel transpile dependencies
    transpile: ['vue-router-tab']
  }
}
